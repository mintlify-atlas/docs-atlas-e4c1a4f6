---
title: 'List Roles'
api: 'GET /v1beta1/roles'
---

Returns a list of platform-wide roles available in the entire Frontier instance along with their associated permissions.

## Query Parameters

<ParamField query="state" type="string">
  Filter roles by state
  
  Values: `enabled`, `disabled`
</ParamField>

<ParamField query="scopes" type="array">
  Filter roles by scopes. Can specify multiple scopes.
</ParamField>

## Response

<ResponseField name="roles" type="array">
  Array of role objects
  
  <Expandable title="properties">
    <ResponseField name="id" type="string">
      Unique identifier for the role
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Role name (unique within organization)
    </ResponseField>
    
    <ResponseField name="title" type="string">
      Human-readable title for the role
    </ResponseField>
    
    <ResponseField name="permissions" type="array">
      Array of permission slugs associated with this role
      
      Example: `["app_organization_update", "app_organization_delete"]`
    </ResponseField>
    
    <ResponseField name="scopes" type="array">
      Array of scope strings used for filtering roles
    </ResponseField>
    
    <ResponseField name="state" type="string">
      Current state of the role
      
      Values: `enabled`, `disabled`
    </ResponseField>
    
    <ResponseField name="org_id" type="string">
      Organization ID this role belongs to (platform roles use a special ID)
    </ResponseField>
    
    <ResponseField name="metadata" type="object">
      Additional metadata for the role
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      Timestamp when the role was created
      
      Format: `date-time`
    </ResponseField>
    
    <ResponseField name="updated_at" type="string">
      Timestamp when the role was last updated
      
      Format: `date-time`
    </ResponseField>
  </Expandable>
</ResponseField>

<CodeGroup>
```bash cURL
curl -X GET 'https://frontier.example.com/v1beta1/roles?state=enabled' \
  -H 'Authorization: Bearer <token>'
```

```javascript JavaScript
const response = await fetch('https://frontier.example.com/v1beta1/roles?state=enabled', {
  method: 'GET',
  headers: {
    'Authorization': 'Bearer <token>'
  }
});

const data = await response.json();
```

```python Python
import requests

response = requests.get(
    'https://frontier.example.com/v1beta1/roles',
    params={'state': 'enabled'},
    headers={'Authorization': 'Bearer <token>'}
)

data = response.json()
```
