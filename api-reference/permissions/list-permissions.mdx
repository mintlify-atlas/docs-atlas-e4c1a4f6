---
title: 'List Permissions'
api: 'GET /v1beta1/permissions'
---

Returns a list of all permissions available in the Frontier instance.

## Response

<ResponseField name="permissions" type="array">
  Array of permission objects
  
  <Expandable title="properties">
    <ResponseField name="id" type="string">
      Unique identifier for the permission
    </ResponseField>
    
    <ResponseField name="name" type="string">
      Permission name (verb part)
    </ResponseField>
    
    <ResponseField name="title" type="string">
      Human-readable title for the permission
    </ResponseField>
    
    <ResponseField name="namespace" type="string">
      Namespace of the permission (service/resource format)
    </ResponseField>
    
    <ResponseField name="key" type="string">
      Permission path key composed of three parts: `service.resource.verb`
      
      Example: `compute.instance.get`
    </ResponseField>
    
    <ResponseField name="metadata" type="object">
      Additional metadata for the permission
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      Timestamp when the permission was created
      
      Format: `date-time`
    </ResponseField>
    
    <ResponseField name="updated_at" type="string">
      Timestamp when the permission was last updated
      
      Format: `date-time`
    </ResponseField>
  </Expandable>
</ResponseField>

<CodeGroup>
```bash cURL
curl -X GET 'https://frontier.example.com/v1beta1/permissions' \
  -H 'Authorization: Bearer <token>'
```
