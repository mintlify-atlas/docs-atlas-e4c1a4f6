---
title: 'Create Group'
openapi: 'POST /v1beta1/organizations/{org_id}/groups'
---

Create a new group within an organization. The authenticated user automatically becomes the owner of the newly created group.

## Path Parameters

<ParamField path="org_id" type="string" required>
  The unique identifier of the organization where the group will be created.
</ParamField>

## Body Parameters

<ParamField body="name" type="string">
  Unique URL-friendly identifier for the group within the organization. Must be unique within the organization.
  
  If not provided but `title` is provided, the name will be auto-generated from the title by converting it to a slug format.
</ParamField>

<ParamField body="title" type="string" required>
  Display name of the group. This is the human-readable name shown in the UI.
</ParamField>

<ParamField body="metadata" type="object">
  Arbitrary metadata to associate with the group as key-value pairs. Can be used to store additional information about the group.
  
  Example: `{"department": "engineering", "cost_center": "1234"}`
</ParamField>

## Response

<ResponseField name="group" type="object">
  The created group object.
  
  <Expandable title="Group Properties">
    <ResponseField name="id" type="string">
      Unique identifier for the newly created group.
    </ResponseField>
    
    <ResponseField name="name" type="string">
      The URL-friendly identifier of the group.
    </ResponseField>
    
    <ResponseField name="title" type="string">
      Display title of the group.
    </ResponseField>
    
    <ResponseField name="org_id" type="string">
      ID of the organization this group belongs to.
    </ResponseField>
    
    <ResponseField name="metadata" type="object">
      Metadata associated with the group.
    </ResponseField>
    
    <ResponseField name="created_at" type="timestamp">
      Timestamp when the group was created.
    </ResponseField>
    
    <ResponseField name="updated_at" type="timestamp">
      Timestamp when the group was last updated.
    </ResponseField>
    
    <ResponseField name="members_count" type="integer">
      Number of members in the group. Will be 1 initially (the creator).
    </ResponseField>
  </Expandable>
</ResponseField>
