---
title: 'Update Group'
openapi: 'PUT /v1beta1/groups/{id}'
---

Update an existing group's details. You can modify the group's name, title, and metadata.

## Path Parameters

<ParamField path="id" type="string" required>
  The unique identifier of the group to update.
</ParamField>

## Query Parameters

<ParamField query="org_id" type="string" required>
  The organization ID that the group belongs to. Required for authorization.
</ParamField>

## Body Parameters

<ParamField body="name" type="string">
  New URL-friendly identifier for the group. Must be unique within the organization.
  
  If you change the name, any existing references using the old name will need to be updated.
</ParamField>

<ParamField body="title" type="string">
  New display name for the group.
</ParamField>

<ParamField body="metadata" type="object">
  Updated metadata for the group. This completely replaces the existing metadata.
  
  To remove metadata, pass an empty object `{}`.
</ParamField>

## Response

<ResponseField name="group" type="object">
  The updated group object.
  
  <Expandable title="Group Properties">
    <ResponseField name="id" type="string">
      Unique identifier for the group (unchanged).
    </ResponseField>
    
    <ResponseField name="name" type="string">
      The updated URL-friendly identifier of the group.
    </ResponseField>
    
    <ResponseField name="title" type="string">
      The updated display title of the group.
    </ResponseField>
    
    <ResponseField name="org_id" type="string">
      ID of the organization this group belongs to.
    </ResponseField>
    
    <ResponseField name="metadata" type="object">
      The updated metadata associated with the group.
    </ResponseField>
    
    <ResponseField name="created_at" type="timestamp">
      Timestamp when the group was created (unchanged).
    </ResponseField>
    
    <ResponseField name="updated_at" type="timestamp">
      Timestamp when the group was last updated. Will reflect the current update time.
    </ResponseField>
    
    <ResponseField name="members_count" type="integer">
      Number of members in the group.
    </ResponseField>
  </Expandable>
</ResponseField>

import (

)

    "metadata": map[string]string{
        "department": "engineering",

    },
}

req, _ := http.NewRequest(
    "PUT",
    "https://frontier.example.com/v1beta1/groups/grp_123456789?org_id=org_123",
    bytes.NewBuffer(body),
)
req.Header.Set("Authorization", "Bearer YOUR_ACCESS_TOKEN")
req.Header.Set("Content-Type", "application/json")

client := &http.Client{}
resp, _ := client.Do(req)
